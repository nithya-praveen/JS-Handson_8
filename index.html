Q1. You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string.

A1.function RemoveDuplicate( STR,N)
{
var s = new Set();
for (var i = 0; i < N ; i++)
s.add(STR[i]);
for (const v of s)
{
document.write(v);
}
}
var STR = "abcadeecfb";
var N = STR.length;
RemoveDuplicate(STR, N);

Q2. You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)

A2.function CountAlph(STR,N)
{
var d = new Map();
STR.split('').forEach(N =>
{
if(d.has(N))
{
d.set(N, d.get(N)+1);
}
else
d.set(N, 1);
});
STR.split('').forEach(N =>
{
if(d.has(N) && d.get(N)!=0)
{
console.log( N +"="+ d.get(N) + "");
d.set(N, 0);
}
});
}
CountAlph("abcadeecfb");
